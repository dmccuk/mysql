CREATE TABLE shirts 
  ( 
     shirt_id INT NOT NULL AUTO_INCREMENT, 
     article   VARCHAR(100), 
     colour  VARCHAR(100), 
     shirt_size CHAR,
     last_worn  INT, 
     PRIMARY KEY (shirt_id) 
  );
